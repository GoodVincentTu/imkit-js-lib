    import io from 'socket.io-client';
import lib from '../lib';
/**
 * @import * from '@/src/typedef';
 */

export default class IMKitSocket {
  /**
   * 建立 Socket 物件
   * @param {SocketConfig} config
   */
  constructor(config) {
    /** @type {SocketConfig} */
    this.config = config;

    /**
     * 上一次收到 ping 的時間
     * @private
     * @type {Number}
     */
    this.pingTime = 0;

    /**
     * 5 秒 ping 一次
     * @private
     * @type {Number}
     */
    this.pingDelay = 6000;

    /**
     * @private
     * @type {Boolean}
     */
    this.connectStatus = false;

    /**
     * @private
     * @type {SocketIOClient.Socket}
     */
    this.socket = io(this.config.domain, {});
    this.socket.connect();

    setInterval(() => {
      if (this.connectStatus) {
        if (Date.now() - this.pingTime > this.pingDelay) {
          this.disconnect();
        }
      } else {
        if (Date.now() - this.pingTime < this.pingDelay) {
          this.onConnect();
        }
      }
    }, this.pingDelay);

    this.socket.on('ping', () => {
      this.pingTime = Date.now();
    });

    this.socket.on('connect', () => {
      this.onConnect();
    });

    this.socket.on('disconnect', () => {
      this.disconnect();
    });

    this.socket.on('chat message', data => {
      if (config.onReceiveMessage) {
        config.onReceiveMessage(
          lib.message.decrypt(
            data,
            this.config.privateKey,
            this.config.clientId
          )
        );
      } else {
        console.log(data);
      }
    });

    this.socket.on('room', data => {
      if (config.onReceiveRoom) {
        config.onReceiveRoom(data);
      } else {
        console.log(data);
      }
    });

    this.socket.on('typing', data => {
      if (config.onReceiveTyping) {
        config.onReceiveTyping(data);
      } else {
        console.log(data);
      }
    });

    this.socket.on('lastRead', data => {
      if (config.onReceivelastRead) {
        config.onReceivelastRead(data);
      } else {
        console.log(data);
      }
    });
  }

  /**
   * @private
   */
  onConnect() {
    this.pingTime = Date.now();
    this.connectStatus = true;
    if (this.config.onConnect) {
      this.config.onConnect();
    } else {
      console.log('Default connect success');
    }
  }

  /**
   * @private
   */
  disconnect() {
    this.pingTime = 0;
    this.connectStatus = false;
    if (this.config.onDisconnect) {
      this.config.onDisconnect();
    } else {
      console.log('Default disconnect success');
    }
  }

  /**
   * chatIn
   * @returns {Promise<{client: String}|Error>}
   */
  chatIn() {
    return new Promise((resolve, reject) => {
      this.socket.emit('auth', this.config.token, res => {
        if (res.RC === 0) {
          resolve(res.result);
        } else {
          this.onError(res);
          reject(new Error(`[IMKit WebSocket] ${res.RM}`));
        }
      });
    });
  }

  /**
   * Send message
   * @param {String} roomId
   * @param {MessageInfo} message
   * @returns {Promise<MessageData|Error>}
   */
  sendMessage(roomId, message) {
    return new Promise((resolve, reject) => {
      let data = Object.assign(message, {
        room: roomId
      });

      this.socket.emit('chat message', data, res => {
        if (res.RC === 0) {
          resolve(res.result);
        } else {
          this.onError(res);
          reject(new Error(`[IMKit WebSocket] ${res.RM}`));
        }
      });
    });
  }

  /**
   * Typing
   * @param {String} roomId
   * @param {String} text
   */
  typing(roomId, text) {
    this.socket.emit('typing', roomId, text);
  }

  /**
   * @private
   * @param {Object} error
   */
  onError(error) {
    this.config.events.onAjaxError(error);
  }
}
