    import axios from 'axios';
/**
 * @import * from '@/src/typedef';
 */

export default class File {
  /**
   * @private
   * @param {Api} api
   */
  constructor(api) {
    /**
     * @private
     * @type {Api}
     * */
    this.api = api;
    /**
     * @private
     * @type {Boolean}
     */
    this.isAdmin = false;
  }

  /**
   * Upload File
   * @param {String} bucketName
   * @param {File} file
   * @param {Object} extra
   * @param {Function} progress
   * @returns {Promise<String|Error>}
   */
  upload(bucketName, file, extra = {}, progress = e => {}) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/files/${encodeURIComponent(
        bucketName
      )}`;
      let data = new FormData();
      let headers = this.api.getHeader(this.isAdmin);

      data.append('extra', JSON.stringify(extra));
      data.append('file', file);
      // headers = Object.assign(headers, data.getHeaders());

      axios
        .post(url, data, {
          headers: headers,
          onUploadProgress: progress
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(
                `${this.api.config.domain}/files/${bucketName}/${
                  res.data.result.id
                }`
              );
            } else {
              resolve(res.data);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Get File metadata
   * @param {String} bucketName
   * @param {String} fileId
   * @returns {Promise<{AcceptRanges: String, LastModified: String, ContentLength: Number, ETag: String, ContentType: String, Metadata: Object[], extra: Object[]|Error>}
   */
  getMetadata(bucketName, fileId) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/files/${encodeURIComponent(
        bucketName
      )}/${fileId}/metadata`;
      let data = {};
      let headers = this.api.getHeader(this.isAdmin);

      axios
        .get(url, {
          params: data,
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Downliad File
   * @param {String} bucketName
   * @param {String} fileId
   * @param {Object} appendHeaders
   * @param {Function} progress
   */
  download(bucketName, fileId, appendHeaders = {}, progress = e => {}) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/files/${encodeURIComponent(
        bucketName
      )}/${fileId}`;
      let data = {
        signedUrlOnly: 1
      };
      let headers = this.api.getHeader(this.isAdmin);

      headers = Object.assign(headers, appendHeaders);

      axios
        .get(url, {
          params: data,
          headers: headers
        })
        .then(
          async res => {
            if (res.status >= 200 && res.status < 300) {
              try {
                let resFile = await this.downloadFileFormRealUrl(
                  res.data.result.url,
                  appendHeaders,
                  progress
                );
                resolve(resFile);
              } catch (e) {
                reject(e);
              }
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }
  /**
   * Downliad File
   * @private
   * @param {String} url
   * @param {Object} appendHeaders
   * @param {Function} progress
   */
  downloadFileFormRealUrl(url, appendHeaders = {}, progress = e => {}) {
    return new Promise((resolve, reject) => {
      let data = {};
      let headers = this.api.getHeader(this.isAdmin);

      headers = Object.assign(headers, appendHeaders);

      axios
        .get(url, {
          params: data,
          headers: headers,
          onDownloadProgress: progress,
          responseType: 'arraybuffer'
        })
        .then(
          async res => {
            if (res.status >= 200 && res.status < 300) {
              resolve(res);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }
}
