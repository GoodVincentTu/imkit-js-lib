    import axios from 'axios';
import Room from '../room';
/**
 * @import * from '@/src/typedef';
 */

export default class Admin {
  /**
   * @private
   * @param {Api} api
   */
  constructor(api) {
    /**
     * @private
     * @type {ApiConfig}
     * */
    this.api = api;
    /**
     * @private
     * @type {Boolean}
     */
    this.isAdmin = true;
    /** @type {Room} */
    this.room = new Room(this.api);
    this.room.isAdmin = this.isAdmin;
  }

  /**
   * Get clients list
   * @param {Number} [limit = 10]
   * @param {{nickname: Object}} [q = {}]
   * @returns {Promise<ClientData[]|Error>}
   */
  getClients(limit = 10, q = {}) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/admin/clients`;
      let data = {
        limit,
        q
      };
      let headers = this.api.getHeader(this.isAdmin);

      axios
        .get(url, {
          params: data,
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Update or create client
   * @param {Object} clientInfo
   * @param {String} [clientInfo._id]
   * @param {String} [clientInfo.avatarUrl] 大頭網址
   * @param {String} [clientInfo.email]
   * @param {String} [clientInfo.nickname] 暱稱
   * @param {String} [clientInfo.publicKey] 訊息加密公鑰
   * @returns {Promise<ClientData|Error>}
   */
  updateClient(clientInfo) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/admin/clients`;
      let data = clientInfo;
      let headers = this.api.getHeader(this.isAdmin);

      axios
        .post(url, data, {
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Bind token to client
   * @param {String} id Client id
   * @param {String} token
   * @returns {Promise<TokenData|Error>}
   */
  bindToken(id, token) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/admin/clients/${id}/token/`;
      let data = {
        token
      };

      let headers = this.api.getHeader(this.isAdmin);

      axios
        .put(url, data, {
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Revoke token to client
   * @param {String} id Client id
   * @param {String} token
   * @returns {Promise<{}|Error>}
   */
  revokeToken(id, token) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/admin/clients/${id}/token/`;
      let data = {
        token
      };

      let headers = this.api.getHeader(this.isAdmin);

      axios
        .delete(url, {
          data: data,
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }
