    import axios from 'axios';
/**
 * @import * from '@/src/typedef';
 */

export default class Me {
  /**
   * @private
   * @param {Api} api
   */
  constructor(api) {
    /**
     * @private
     * @type {Api}
     * */
    this.api = api;
    /**
     * @private
     * @type {Boolean}
     */
    this.isAdmin = false;
  }

  /**
   * Get my info
   * @returns {Promise<ClientData|Error>}
   */
  getInfo() {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/me`;
      let data = {};
      let headers = this.api.getHeader(this.isAdmin);

      axios
        .get(url, {
          params: data,
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }

  /**
   * Update my info
   * @param {Object} info
   * @param {String} [info.avatarUrl] 大頭網址
   * @param {String} [info.email]
   * @param {String} [info.nickname] 暱稱
   * @returns {Promise<ClientData|Error>}
   */
  updateInfo(info) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/me`;
      let data = info;
      let headers = this.api.getHeader(this.isAdmin);

      axios
        .post(url, data, {
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              resolve(res.data.result);
