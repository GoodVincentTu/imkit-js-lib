  /**
   * Get rooms list
   * @param {Number} [limit = 20] 最大回傳筆數
   * @param {Number} [skip = 0] 從第幾筆開始
   * @param {{members: String}} [q = {}]
   * @param {String} [sort] 排序方式
   * @returns {Promise<{totalCount: Number, data: RoomData[]}|Error>}
   */
  getRoomsList(limit = 20, skip = 0, q = {}, sort = null) {
    return new Promise((resolve, reject) => {
      let url = `${this.api.config.domain}/rooms`;
      let data = {
        limit,
        skip,
        q
      };
      let headers = this.api.getHeader(this.isAdmin);

      if (sort) {
        data.sort = sort;
      }
      axios
        .get(url, {
          params: data,
          headers: headers
        })
        .then(
          res => {
            if (res.status >= 200 && res.status < 300 && res.data.RC === 0) {
              // 解密訊息
              if (res.data.result.data) {
                res.data.result.data.forEach(room => {
                  if (room.lastMessage) {
                    room.lastMessage = lib.message.decrypt(
                      room.lastMessage,
                      this.api.config.privateKey,
                      this.api.config.clientId
                    );
                  }
                });
              }

              resolve(res.data.result);
            }
          },
          error => {
            this.api.onError(error);
            reject(error);
          }
        );
    });
  }
